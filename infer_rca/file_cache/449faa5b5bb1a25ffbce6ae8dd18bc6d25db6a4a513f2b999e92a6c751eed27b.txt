import hashlib
import os
import uuid
import traceback
import requests
from fastapi import FastAPI, Request
from auth import verify_user
from starlette.middleware.base import BaseHTTPMiddleware

class InferTracerMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        trace_id = uuid.uuid4().hex[:8] 
        
        try:
            # 1. Execute the actual request
            response = await call_next(request)
            
            # 2. SUCCESS PATH: Send lightweight heartbeat to clear dashboard
            success_payload = {
                "trace_id": trace_id,
                "status": "success",
                "endpoint": request.url.path,
                "event": "200 OK" # For Splunk
            }
            try:
                # Use tiny timeout so it doesn't slow down the target app
                requests.post("http://localhost:5050/ingest-success", json=success_payload, timeout=0.5)
            except Exception:
                pass # Ignore if agent is offline
                
            return response
            
        except Exception as e:
            error_trace = traceback.format_exc()
            extracted_tb = traceback.extract_tb(e.__traceback__)

            file_inventory = {} 
            files_payload = []  
            processed_filepaths = set() # NEW: Track paths properly

            # 1. Loop through every file involved in the crash
            for frame in extracted_tb:
                filepath = frame.filename
                
                # 2. Filter out the noise
                if "site-packages" in filepath or "venv" in filepath or filepath.startswith("<"):
                    continue
                
                clean_filename = os.path.basename(filepath)
                
                # 3. FIX: Deduplicate using the new set
                if filepath in processed_filepaths:
                    continue
                processed_filepaths.add(filepath)

                # 4. Read and Hash (Keep your existing try/except block here)
                try:
                    with open(filepath, "r", encoding="utf-8") as f:
                        content = f.read()
                        
                    # Generate the SHA-256 hash of the file's content
                    file_hash = hashlib.sha256(content.encode('utf-8')).hexdigest()
                    
                    # Store the heavy content locally just in case
                    file_inventory[file_hash] = {
                        "filename": clean_filename,
                        "filepath": filepath,
                        "content": content,
                        "hash": file_hash # Add this line!
                    }
                    
                    # Add the lightweight signature to our initial payload
                    files_payload.append({
                        "filename": clean_filename,
                        "filepath": filepath,
                        "hash": file_hash
                    })
                except Exception:
                    continue # Skip binary or unreadable files silently

            # Determine the primary file that crashed (usually the last one in our filtered list)
            primary_crashed_file = files_payload[-1]["filename"] if files_payload else "Unknown"

            # 5. Build the Lightweight Payload
            log_payload = {
                "trace_id": trace_id,
                "endpoint": request.url.path,
                "error_type": type(e).__name__,
                "traceback": error_trace,
                "primary_crashed_file": primary_crashed_file,
                "files": files_payload # Only sending hashes, NOT the code!
            }
            
            print("\n" + "="*60)
            print(f"[INFER] Crash Intercepted | Trace-ID: {trace_id}")
            print(f"[INFER] Primary File: {primary_crashed_file}")
            print("="*60 + "\n")
            
            # 6. The Handshake Protocol
            try:
                # Step A: Send the lightweight hash payload
                response = requests.post("http://localhost:5050/ingest", json=log_payload, timeout=5)
                
                if response.status_code == 200:
                    agent_data = response.json()
                    missing_hashes = agent_data.get("missing_hashes", [])
                    
                    # Step B: If the Agent is missing files, upload them
                    if missing_hashes:
                        print(f"[INFER] Agent requested {len(missing_hashes)} missing files. Uploading...")
                        
                        code_payload = {
                            "trace_id": trace_id,
                            "missing_files": [
                                file_inventory[h] for h in missing_hashes if h in file_inventory
                            ]
                        }
                        # Send to a NEW route on the agent designed just for code uploads
                        requests.post("http://localhost:5050/upload-code", json=code_payload, timeout=5)
                        
            except requests.exceptions.RequestException as req_err:
                # Added the actual error print so we can see exactly why it failed if it happens again
                print(f"[INFER-WARNING] Connection failed: {req_err}")
            
            raise e

app = FastAPI()
app.add_middleware(InferTracerMiddleware)

@app.get("/checkout/{user_id}")
async def run_checkout(user_id: str, cart_total: float = 25.00):
    user = verify_user(user_id)

    current_balance = user.get("balance")
    if current_balance < cart_total:
        return {"error": "Insufficient funds"}

    return {"status": "Success"}

if __name__ == "__main__":
    print("Starting target app on port 10000...")
    uvicorn.run(app, host="127.0.0.1", port=10000)
